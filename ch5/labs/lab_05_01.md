
# Lab 05-1

```js 
// 建立購物車
let cart = ['Laptop', 'Mouse', 'Keyboard'];

// 取出元素
console.log(cart[0]);           // ?
console.log(cart[cart.length - 1]); // ?

// 動態擴增
cart[cart.length] = 'Monitor';

// 再加入一個商品
cart.push('USB');

// 更新商品
cart[1] = 'Wireless Mouse';

// 嘗試刪除商品
delete cart[2];

// 印出目前購物車
console.log(cart);
console.log(cart.length);

// 使用 splice 再刪除一個商品
cart.splice(2, 1);

// 最終結果
console.log(cart);
console.log(cart.length);
```
